BLOCK START
SET R_1 0d3
SET R_2 0d1
ADD R_1 R_2 R_3
STORE CARRY
SET R_TRUE 0d2
SET R_FALSE 0d0
SET R_ONE 0d1
CMP CARRY R_ONE R_TRUE R_FALSE R_result
JIE R_result R_TRUE CARRY_Y
JMP CARRY_N

BLOCK CARRY_Y
SETMEMAD 0d1
SET R_MEM 0d1
STORE R_MEM
END

BLOCK CARRY_N
SETMEMAD 0d2
SET R_MEM 0d1
STORE R_MEM
SETMEMAD 0d3
STORE R_MEM
END
